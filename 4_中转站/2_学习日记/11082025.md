# .gitlab/ci_templates/checkstyle.gitlab-ci.yml
stages:
  - validate

checkstyle:
  stage: validate
  image:
    name: gradle@sha256:f30921a2af0cca204583cabc319afcc5b7ca701702c19913815f5eef6f834191
    pull_policy: if-not-present
  allow_failure: true

  cache:
    key:
      prefix: "$CI_PROJECT_ID-$CI_COMMIT_REF_SLUG"
      files:
        - build.gradle
        - '**/build.gradle'
    paths:
      - 'violations-command-line-*.jar'
      - .gradle/
      - '**/.gradle/caches/'
      - '**/.gradle/wrapper/'
    policy: pull-push

  variables:
    GRADLE_USER_HOME: .gradle
    GIT_STRATEGY: fetch
    GIT_DEPTH: 0
  interruptible: true
  timeout: 10m

  script:
    - |-
      export PS4='[$(date +%H:%M:%S)] ${0##*/}:${LINENO}: '
      set -Eeuo pipefail
      set -x
      
      # ------------- 1. locate newest Violations CLI -----
      VCL_REPO="https://github.com/tomasbjerre/violations-command-line.git"
      LATEST_TAG=$(git ls-remote --tags --refs --sort=-v:refname "$VCL_REPO" \
                    | head -n1 | awk -F/ '{print $3}')   # e.g. 3.1.1 :contentReference[oaicite:0]{index=0}
      VCL_VER="${LATEST_TAG#v}"                          # strip optional “v”
      VCL_JAR="violations-command-line-${VCL_VER}.jar"
      VCL_URL="https://repo1.maven.org/maven2/se/bjurr/violations/violations-command-line/${VCL_VER}/${VCL_JAR}"  # directory exists :contentReference[oaicite:1]{index=1}

      if [[ ! -f "$VCL_JAR" ]]; then
        echo "[INFO] Downloading $VCL_JAR …"
        if ! curl --retry 3 --retry-all-errors \
          -fSL --progress-bar "$VCL_URL" -o "$VCL_JAR"; then
          echo "[ERROR] Failed to download $VCL_JAR"
          exit 1         # or handle the failure another way
        fi
      fi
      
      is_valid_commit() {
        git cat-file -e "$1^{commit}" 2>/dev/null
      }

      # ============================================================
      #  pick_diff_base
      #
      #  Determines the appropriate base commit to compare against
      #  for detecting changed Java files in the pipeline.
      #
      #  Strategy:
      #    1. Merge Request pipeline:
      #       → Use CI_MERGE_REQUEST_TARGET_BRANCH_SHA
      #         (target commit of the MR)
      #
      #    2. Branch pipeline with valid CI_COMMIT_BEFORE_SHA:
      #       → Use CI_COMMIT_BEFORE_SHA
      #         (tip of the branch before this push)
      #
      #    3. Branch pipeline without CI_COMMIT_BEFORE_SHA, and
      #       branch has ≥ 2 commits:
      #       → Use HEAD~1
      #         (direct parent of the current commit)
      #
      #    4. First commit on a new branch:
      #       → Use Git’s empty tree SHA (4b825d...)
      #         (treats all files as newly added)
      #
      #  This ensures compatibility with:
      #    - Merge Request pipelines
      #    - Single or multi-commit pushes
      #    - First commits on new branches
      #    - Manual/scheduled pipelines
      # ============================================================
      pick_diff_base() {
        local empty_tree="4b825dc642cb6eb9a060e54bf8d69288fbee4904"
      
        if [[ -n "${CI_MERGE_REQUEST_TARGET_BRANCH_SHA:-}" ]] \
          && is_valid_commit "$CI_MERGE_REQUEST_TARGET_BRANCH_SHA"; then
          echo "$CI_MERGE_REQUEST_TARGET_BRANCH_SHA"
      
        elif [[ -n "${CI_COMMIT_BEFORE_SHA:-}" \
          && "${CI_COMMIT_BEFORE_SHA}" != "$empty_tree" ]] \
          && is_valid_commit "$CI_COMMIT_BEFORE_SHA"; then
          echo "$CI_COMMIT_BEFORE_SHA"
      
        elif git rev-parse --verify HEAD~1 >/dev/null 2>&1; then
          git rev-parse HEAD~1
      
        else
          echo "$empty_tree"      # first commit in repo
        fi
      }
      
      # ------------------ main --------------------
      BASE="$(pick_diff_base)"
      CHANGED="$(git diff --name-only --diff-filter=ACMRTUXB "$BASE" "$CI_COMMIT_SHA" -- ':(glob)**/*.java')"
      echo "Diff base: $BASE"
      echo "Changed files:"
      echo "$CHANGED"
      
      if [ -z "$CHANGED" ]; then
        echo "No changed Java files."
        exit 0
      fi
      
      echo "Changed Java files:"
      echo "$CHANGED"
      
      # Create comma-separated param for Gradle
      files_param=$(printf '%s\n' "$CHANGED" | paste -sd ',' - | sed 's/,$//')
      echo "files_param = $files_param"
      
      # -------- run Checkstyle ----------
      gradle_cmd="gradle"
      [[ -f ./gradlew ]] && gradle_cmd="./gradlew"
      echo "Test 001"
      "${gradle_cmd}" --no-daemon --console=plain --warning-mode=fail --info \
        -Pfiles="${files_param}" checkstyleFiles || checkstyle_exit_code=$?
      echo "Test 000"
      
      # --- 4. Merge all XML outputs ------------------------------------------
      merged_xml="merged-checkstyle.xml"
      if find . -type f -path '*/build/reports/checkstyle/*.xml' \
         -not -name "$merged_xml" | grep -q .; then
        echo "Test 111"
        printf '<?xml version="1.0" encoding="UTF-8"?><checkstyle version="8.0">\n' >"$merged_xml"
        echo "Test 112"
        find . -type f -path '*/build/reports/checkstyle/*.xml' \
        -not -name "$merged_xml" \
        -exec sed -n '/<file /,/<\/file>/p' {} \; >>"$merged_xml"
        echo "Test 113"
        echo '</checkstyle>' >>"$merged_xml"
      fi
      
      echo "Test 1"
      
      merged_xml_dir="build/reports/checkstyle"
      merged_xml_path="$merged_xml_dir/$merged_xml"
      mkdir -p "$merged_xml_dir"
      mv -f "$merged_xml" "$merged_xml_dir/"
      
      # --- 5. Convert to Code-Climate JSON ------------------------------------
      if [[ -s $merged_xml_path ]]; then
        java -jar "$VCL_JAR" \
             -cc gl-code-quality-report.json \
             -v "CHECKSTYLE" "." ".*${merged_xml_path}" "Checkstyle"
      else
        echo "No Checkstyle findings."
      fi
      
      echo "Test 1"

      # -------- HTML paths for easy click-through ----------
      find . -type f -name '*.html' -path '*/reports/*' | sort
      
      if [[ ! -f gl-code-quality-report.json ]]; then
        echo '[]' > gl-code-quality-report.json
      fi
      
      mkdir -p build/reports/checkstyle

      exit "${checkstyle_exit_code:-0}"

  artifacts:
    when: always
    expire_in: 1 week
    paths:
      - '**/build/reports/checkstyle/'
      - gl-code-quality-report.json
    reports:
      junit: 'build/reports/checkstyle/merged-checkstyle.xml'  # For GitLab test tab
      codequality: gl-code-quality-report.json                  # For MR annotations
    exclude:
      - '**/.gradle/**/*'

  rules:
    - if: '$CI_PIPELINE_SOURCE == "merge_request_event"'
      when: on_success
    - if: '$CI_PIPELINE_SOURCE == "push"'
      when: on_success
    - when: never
